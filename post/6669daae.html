Golang之常量与IOTA

发表于2020-04-23|更新于2021-04-28|Golang
|字数总计:231|阅读时长:1分钟|阅读量:

常量

常量: 定义了之后不能修改, 修改会报错．

常量的定义

//常量的声明
const p = 3.1415926
//批量声明常量
const (
	statusOk = 200
	notFound = 404
)

//如果某一行声明后没有赋值, 默认和上一行一样
const (
	n1 = 200
	//n2 & n3 = n1
	n2
	n3
)

IOTA

//iota
const (
	a1 = iota // 0
	a2 = iota // 1
	a3        // 2
)

const (
	b1 = iota // 0
	b2 = iota // 1
	_  = iota // 匿名变量: 2
	b3 = iota // 3
)

//插队
const (
	c1 = iota // 0
	c2 = 100  // 100
	c3 = iota //2
	c4        //3
)

多个变量声明在同一行

//多个变量声明在一行
const (
	d1, d2 = iota + 1, iota + 2 // d1: 1, d2: 2
	d3, d4 = iota + 1, iota + 2 // d3: 2, d4: 3
)

定义数量级

package main

import "fmt"
const (
	_  = iota             // 0
	KB = 1 << (10 * iota) // 1024 << 往左移十位
	MB = 1 << (10 * iota)
	GB = 1 << (10 * iota)
	TB = 1 << (10 * iota)
)

func main(){
    fmt.Println("KB: ", KB)
	fmt.Println("MB: ", MB)
	fmt.Println("GB: ", GB)
	fmt.Println("TB: ", TB)
}

输出结果:

KB:  1024
MB:  1048576
GB:  1073741824
TB:  1099511627776
