学习 golang 运算符

var (
		a = 5
		b = 2
	)
fmt.Println(a + b)
fmt.Println(a - b)
fmt.Println(a * b)
fmt.Println(a / b)
fmt.Println(a % b)
a++ //单独的语句, 不能有返回值
b-- //单独的语句, 不能有返回值
fmt.Println(a)

关系运算符

var (
		a = 5
		b = 2
	)
fmt.Println(a == b) //go 语言是强类型, 相同类型变量才能比较
fmt.Println(a != b)
fmt.Println(a > b)
fmt.Println(a < b)
fmt.Println(a >= b)
fmt.Println(a <= b)

逻辑运算符

var age = 22
if age > 18 && age < 60 {
	fmt.Println("打工仔")
} else {
	fmt.Println("神仙")
}
if age < 18 || age > 60 {
	fmt.Println("生活美滋滋")
} else {
	fmt.Println("苦逼一个")
}
//not 取反
var isMarried = false
fmt.Println(isMarried)
fmt.Println(!isMarried)

位运算符

//位运算符: 针对二进制数
// 5的二进制表示: 101
// 2的二进制表示: 10
// &: 按位与
fmt.Println(5 & 2) //000 两位都是1 才为1
// |: 按位或 (两位有一个为1 就为1)
fmt.Println(5 | 2) // 111 = 7
// ^: 按位异或 (两位不一样则为1)
fmt.Println(5 ^ 2) //111
// <<: 将二进制位往左移指定位数
fmt.Println(5 << 1)  //1010 => 10
fmt.Println(1 << 10) //10000000000 => 1024
//将二进制数往右移指定位数
fmt.Println(5 >> 2) //1 => 1
var m = int8(1)     //只能存8位
fmt.Println(m << 10)

赋值运算符

//赋值运算, 用来给变量赋值的
var x int
x = 10
x += 1 //x = x + 1
x -= 1 //x = x - 1
x *= 2 //x = x * 2
x /= 2 //x = x / 2
x %= 2 //x = x % 2
x <<= 2 //x = x << 2
x >>= 2 //x = x >> 2
x &= 2  //x = x & 2
x |= 2  //x = x | 2
x ^= 4  //x = xx ^ 4
fmt.Println(x)
