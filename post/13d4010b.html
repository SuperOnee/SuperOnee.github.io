Golang Framework Gin

Gin 框架的介绍:

Gin 是一个用 Go 语言编写的 web 框架. 它是一个类似于 martini 但拥有更好性能的 API 框架, 由于使用了 httprouter, 速度提高了近 40 倍. 如果你是性能和高效的追求者, 你会爱上 Gin. Go 世界里最流行的 Web 框架, Github 上有 36K+star. 基于 httprouter 开发的 Web 框架. 中文文档齐全，简单易用的轻量级框架.

初识 Gin 框架, 老规矩肯定是先来一个 hello world

首先第一步肯定是先安装 Gin 框架: go get -u github.com/gin-gonic/gin

然后就是开始代码的编写啦
代码如下:

package main

import "github.com/gin-gonic/gin"

func main() {
   app := gin.Default()
   app.GET("/", func(context *gin.Context) {
      //其实gin.H 就是 map[string]interface{} 的缩写, 使用时简单明了
      context.JSON(http.StatusOK,gin.H{
         "message":"Hello World",
      })
   })

   app.Run() // listen and serve on 0.0.0.0:8080 (for windows "localhost:8080") 如果不指定端口号的话默认是为8080端口
}
//gin初始化的方式有gin.Default() 和 gin.New(), 如果不想要一些中间件的话可以用gin.New来初始化Engine
//下面是gin.Default的源码

// Default returns an Engine instance with the Logger and Recovery middleware already attached.
func Default() *Engine {
	debugPrintWARNINGDefault()
	engine := New()
	engine.Use(Logger(), Recovery())
	return engine
}

Gin 的路由

路由

Get 获取 Get 参数 name := context.Query("name")

func main() {
   app := gin.Default()

   app.GET("/", func(context *gin.Context) {
      context.JSON(http.StatusOK,gin.H{
         "message":"Hello World",
      })
   })

   app.GET("/pps",pps) // /pps?name=?&password=?

   app.Run()
}


func pps(context *gin.Context) {
   name := context.Query("name")
   pwd := context.Query("password")

   context.JSON(http.StatusOK,gin.H{
      "name":name,
      "pwd":pwd,
   })
}

路由参数

app.GET("/name/:name/:age",hello)

func hello(context *gin.Context) {
   name := context.Param("name")
   age := context.Param("age")
   context.JSON(http.StatusOK,gin.H{
      "name":name,
      "age":age,
   })

POST

func ppos(context *gin.Context) {
   //context.DefaultPostForm("type", "alert") //可以使用DefaultPostForm指定一个默认的key和value发送到表单
   name := context.PostForm("name") //将name作为key发送到表单
   password := context.PostForm("password") //将password作为key发送到表单
   context.JSON(http.StatusOK,gin.H{
      "name":name,
      "password":password,
   })
}

JSON

func jsson(context *gin.Context) {
   body := context.Request.Body
   defer body.Close()
   type User struct {
      User string `json:"user"`
      Password string `json:password`
   }
    //将请求体读出来用json.Unmarshal反序列化操作写到user struct(user结构体)里
   bytes, _ := ioutil.ReadAll(body)
   user := &User{}
   json.Unmarshal(bytes,user)
   context.JSON(http.StatusOK,user)
}

JSON 参数绑定
注意: 要设置type:application/json才能接收到!

func jsonBinding(context *gin.Context) {
	type user struct {
		User     string `json:"user"`
		Password string `json:"password"`
	}
	u := &user{
		User:     "帅比",
		Password: "123",
	}
	err := context.Bind(u)
	if err != nil {
		fmt.Println(err.Error())
	}
	fmt.Println(u.User)
	fmt.Println(u.Password)
	context.JSON(http.StatusOK, u)
}

路由的分组

package main

import "github.com/gin-gonic/gin"

func main() {
   app := gin.Default()

   // version one (版本1)
   v1 := app.Group("/v1")
   {
      v1.GET("/hello",v1Fun)
   }

   v2 := app.Group("/v2")
   {
      v2.GET("/hello",v2Fun)
      admin := v2.Group("/admin")
      {
         admin.GET("/home",adminHome)
      }
   }

   app.Run() //default port equals to 8080
}

func v1Fun(context *gin.Context)  {
   context.JSON(http.StatusOk,gin.H{
      "hello":"world",
   })
}

func v2Fun(context *gin.Context) {
   context.String(http.StatusOk,"Hello World")
}

func adminHome(context *gin.Context) {
   context.String(http.StatusOk,"this is the home page")
}

Middleware(中间件)

Gin 的 Middleware 和 iris 的 Middleware 使用上面差不多
(主要差别在于 Gin 默认是放行,iris 默认是不放行)

func main() {
   app := gin.Default()
   app.Use(MiddleMain) // 全局
   //admin := app.Group("/admin")
   //admin.Use() // 制定分组
   app.GET("/",MiddleAuth,container,MiddleAfter)
   app.GET("/middle",before,content,after)

   app.Run() //default port 8080
}

func MiddleMain(context *gin.Context)  {
   fmt.Println("Main Middleware")
}

func before(context *gin.Context) {
   fmt.Println("before")
   context.Next()
   fmt.Println("before...........")
}

func after(context *gin.Context) {
   fmt.Println("after")
}

func content(context *gin.Context) {
   fmt.Println("content")
   context.Abort() // 不放行
}

context.Next() 放行 context.Abort() 拦截

数据绑定

绑定 Url 查询参数 (Only Bind Query String)

使用context.ShouldBindQuery()方法

type User struct {
   User string `form:"user"`        // 注意tag要form
   Password string `form:"password"`
}

func main() {
   app := gin.Default()
   app.GET("/test",getlog)
   app.Run() //default port 8080
}

func getlog(context *gin.Context)  {
   var user User
   if context.ShouldBindQuery(&user) == nil { // 绑定
      log.Println(user.User)
      fmt.Println(context.Query("user"))
      log.Println(user.Password)
   }
   context.JSON(http.StatusOk,user)
}

绑定 url 查询参数和 POST 参数

使用 c.ShouldBind 方法, 会根据 content-type 类型, 优先匹配 JSON 或者 XML,之后才是 Form.

type User struct {
   User string `form:"user"`
   Password string `form:"password"`
}

func main() {
   app := gin.Default()
   app.GET("/test",getlog)
   app.POST("/logo",logo)
   app.Run()
}
func logo(context *gin.Context) {
   user := &User{}
   context.ShouldBind(user)
   context.JSON(http.StatusOk,user)
}

文件的上传

单文件上传

注意: 前端的的表单要指定 enctype=”multipart/form-data”, 文件的 name 指定成 file

func main() {
   app := gin.Default()
   app.POST("/upload", func(context *gin.Context) {
      app.MaxMultipartMemory = 100 << 20 // 设置最大上传大小为100M
      header, _ := context.FormFile("file")

      path := "./file/" + header.Filename // 上传存储到的地址
      fmt.Println(path)
      err := context.SaveUploadedFile(header, path)
      if err != nil {
         fmt.Println(err.Error())
      }

      log.Println(header.Filename," ",header.Size," ",header.Header)
      context.JSON(http.StatusOk,gin.H{
         "fileName":header.Filename,
      })
   })

   app.Run()
}
多文件的上传
func main() {
   app := gin.Default()
   app.POST("/uploads", func(context *gin.Context) {
      form, _ := context.MultipartForm()
      files := form.File["upload[]"]
      for _,file := range files { //循环读取文件
         log.Println(file.Filename)
         context.SaveUploadedFile(file,"./file/"+file.Filename) // 文件夹需要创建
      }
   })
   app.Run()
}

Validate 数据验证

绑定 JSON(ShouldBindJSON)

type Login struct {
   User     string `form:"user" json:"user" xml:"user"  binding:"required"`
   Password string `form:"password" json:"password" xml:"password" binding:"required"` //结构体field标签指定`binding:"required"` 如果没有数据就会报错
}

func main() {
	router := gin.Default()

	router.POST("/loginJSON", func(context *gin.Context) {
		var json Login
		if err := context.ShouldBindJSON(&json); err != nil {
			context.JSON(http.StatusBadRequest, gin.H{"error": "Login information is not complete"})
			return
		}

		if json.User != "admin" || json.Password != "123" {
			context.JSON(http.StatusUnauthorized, gin.H{"status": "unauthorized"})
			return
		}

		context.JSON(http.StatusOK, gin.H{"status": "you are logged in"})
	})

	router.Run()
}

绑定 XML(ShouldBindXML)

package main

import (
	"github.com/gin-gonic/gin"
	"net/http"
)

type Login struct {
	User     string `form:"user" json:"user" xml:"user"  binding:"required"`
	Password string `form:"password" json:"password" xml:"password" binding:"required"` //结构体field标签指定`binding:"required"` 如果没有数据就会报错
}

func main() {
	router := gin.Default()
	// Example for binding XML (
	// <?xml version="1.0" encoding="UTF-8"?>
	// <root>
	//    <user>user</user>
	//    <password>123</user>
	// </root>)
	router.POST("/loginXML", func(context *gin.Context) {
		var xml Login
		if err := context.ShouldBindXML(&xml); err != nil {
			context.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
			return
		}
		if xml.User != "admin" || xml.Password != "123" {
			context.JSON(http.StatusUnauthorized, gin.H{"status": "unauthorized"})
			return
		}

		context.JSON(http.StatusOK, gin.H{"status": "you are logged in"})
	})
	router.Run()
}

绑定 FORM 表单

context.ShouldBind(&form)

package main

import (
	"github.com/gin-gonic/gin"
	"net/http"
)

type Login struct {
	User     string `form:"user" json:"user" xml:"user"  binding:"required"`
	Password string `form:"password" json:"password" xml:"password" binding:"required"` //结构体field标签指定`binding:"required"` 如果没有数据就会报错
}

func main() {
	router := gin.Default()
	// Example for binding a HTML form (user=admin&password=123)
	router.POST("/loginForm", func(context *gin.Context) {
		var form Login
		// This will infer what binder to use depending on the content-type header.
		if err := context.ShouldBind(&form); err != nil {
			context.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
			return
		}

		if form.User != "admin" || form.Password != "123" {
			context.JSON(http.StatusUnauthorized, gin.H{"status": "unauthorized"})
			return
		}

		context.JSON(http.StatusOK, gin.H{"status": "you are logged in"})
	})

	router.Run()
}

只限查询参数

BindQuery\ShouldBindQuery 函数只绑定查询参数,而不绑定 POST 数据.
我现在只 BindQuery 来试验, ShouldBindQuery 我测试输出结果一样

package main

import (
	"log"
	"net/http"
)
import "github.com/gin-gonic/gin"

type Person struct {
	Name    string `form:"name"`
	Address string `form:"address"`
}

func main() {
	route := gin.Default()
	//Any registers a route that matches all the HTTP methods. route.Any可以匹配所有的http请求方式.
	route.Any("/testing", startPage)
	route.Run()
}
func startPage(context *gin.Context) {
	var person Person
	if context.BindQuery(&person) == nil {
		log.Println("====== Only Bind Query String ======")
		log.Println(person.Name)
		log.Println(person.Address)
	}
	context.String(http.StatusOK, "Success")
}
