学习 golang 数组

数组
必须指定存放的元素的类型和容量(长度)
数组的长度是数组类型的一部分

定义数组

var a1 [3]bool //其中3为数组的长度
var a2 [4]bool //bool 数组值类型为[true, false]
fmt.Printf("a1: %T a2: %T\n", a1, a2) //占位符%T为获取类型

//输出结果:
a1: [3]bool a2: [4]bool

数组的初始化

var a1 [3]bool //默认值为false
var a2 [4]bool
//不初始化都为默认值
fmt.Println(a1, a2)
// 1. 初始化方式1
a1 = [3]bool{true, false, true}
fmt.Println(a1)

// 2. 初始化方式2
array := [...]int{1, 2, 3, 4, 5, 6, 7, 8, 9}
fmt.Println(array)

// 3. 初始化方式3
array2 := [5]int{0: 1, 4: 2} //根据索引初始化, 初始化索引为0的和4的, 其他的都为默认值0
fmt.Println(array2)

//输出结果:
[false false false] [false false false false]
[true false true]
[1 2 3 4 5 6 7 8 9]
[1 0 0 0 2]

数组的遍历

cities := [...]string{"北京", "上海", "深圳"}
// 1. 根据索引遍历
for i := 0; i < len(cities); i++ {
	fmt.Println(cities[i])
}
// 2. for range遍历
for i, v := range cities {
	fmt.Println(i, v)
}

//输出结果:
北京
上海
深圳
0 北京
1 上海
2 深圳

多维数组和遍历

//多维数组
// [[1 2] [3 4] [5 6]]
var arr [3][2]int
arr = [3][2]int{
	[2]int{1, 2},
	[2]int{3, 4},
	[2]int{5, 6},
}
//遍历多维数组
for _, v1 := range arr {
	for _, v2 := range v1 {
		fmt.Println(v2)
	}
}
fmt.Println("直接输出多维数组:", arr)

//输出结果:
1
2
3
4
5
6
直接输出多维数组: [[1 2] [3 4] [5 6]]

==和!=操作符

//数组是值类型, 数组支持 == 和 != 操作符
arr1 := [3]int{1, 2, 3}
arr2 := arr1
arr2[0] = 100
fmt.Println("arr1: ", arr1)
fmt.Println("arr2: ", arr2)
fmt.Println(arr1 == arr2)
fmt.Println(arr1 != arr2)

//输出结果:
arr1:  [1 2 3]
arr2:  [100 2 3]
false
true

练习题

//练习题1: 计算int数组所有元素的和
var sum int = 0
	testArray := [...]int{1, 3, 5, 7, 8}
	for _, v := range testArray {
		sum += v
	}
	fmt.Println(sum)

//输出结果:
24

//练习题2 找出数组中相加为8的两个数的下标
for i := 0; i < len(testArray); i++ {
		for j := i + 1; j < len(testArray); j++ {
			if testArray[i]+testArray[j] == 8 {
				fmt.Printf("(%d, %d)", i, j)
			}
		}
	}

//输出结果:
(0, 3)(1, 2)
